/* FUNCTIONS */                            
TESTSELECT: PROCEDURE /* TESTING SELECT */ 
 PARSE ARG NUM CAR .                       
 SELECT                                    
   WHEN NUM < 25 THEN                      
     STATUS = "25%"                        
   WHEN NUM < 50 THEN                      
     DO                                    
       IF CAR = 'RAD' THEN                 
          STATUS = 'DOPE'                  
       ELSE                                
          STATUS = 'SICK'                  
     END                                   
   WHEN NUM < 200 THEN DO                  
     NOP                                   
   END                                     
   WHEN NUM < 75 THEN                      
     STATUS = 'SEVENTY FIVE'               
   OTHERWISE                               
     IF CAR = 'RAD' THEN                   
       STATUS = 'COOL'
      STATUS = 'COOL'   
    ELSE                
      STATUS = 'RADICAL'
END                     
RETURN STATUS
