//ControlFlow-specific keywords
fn controlflow(y: Vec<usize>) -> usize {
    let mut x = 0;
    while x < 10 {
        if x > 5 {
            x = 1;
        }
        else {
            return 5
        }
    }
    loop {
        x+= 1;
        break;
    }
    for z in y {
        match z {
            1 => continue,
            _ => x = 2,
        }
    }
    x
}
